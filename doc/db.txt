Database : website

Collection "category" :
{
	id: 		    ObjectId,
	categoryId:	    Integer,    paramètre URL
	rank:           Integer,
	title: 		    String,
	description:	String,
	online: 	    Boolean,
	access: 	    String 	    G = guest (tout le monde) / U = user (authentifié)
}

Collection "gallery" :
{
	id: 		    ObjectId,
	categoryId:	    Integer,
	rank:           Integer,
	galleryId:	    Integer,    paramètre URL
    date: 		    String,     format year/month
    title:		    String,
    description:	String,
    thumbnail: 	    String,
    online:		    Boolean,
    access:		    String,
    pictures: 	    [{  thumbnail: 	    String,
                        web:		    String,
                        print:          String,
                        description:	String}
                    ]
}
